Choose your language and copy the code below. Replace YOUR_API_KEY with your actual key.

// Install first: npm install axios const axios = require('axios') const response = await axios.post( 'https://vibecore-07n6.onrender.com/generate', { prompt: 'What is the capital of France?' }, { headers: { 'x-api-key': 'YOUR_API_KEY' } } ) console.log(response.data.response) // Paris console.log('Saved: Rs.' + response.data.saved)
